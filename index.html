return cache.addAll(ASSETS); }) ); }); // Ativação: Elimina caches antigos para garantir que as novas lógicas de data/lixeira apareçam self.addEventListener('activate', (event) => { event.waitUntil( caches.keys().then((keys) => { return Promise.all( keys.filter((key) => key !== CACHE_NAME).map((key) => caches.delete(key)) );
